068/**
069 * Converts the message data received from the wire into a format that Apache Camel processors can consume
070 *
071 * @version
072 */
073@Metadata(label = "eip,transformation")
074@XmlRootElement(name = "unmarshal")
075@XmlAccessorType(XmlAccessType.FIELD)
076public class UnmarshalDefinition extends NoOutputDefinition<UnmarshalDefinition> {
077
078    // TODO: Camel 3.0, ref attribute should be removed as UnmarshalDataFormat is to be used instead
079
080    // cannot use @XmlElementRef as it doesn't allow optional properties
081    @XmlElements({
082        @XmlElement(required = false, name = "avro", type = AvroDataFormat.class),
083        @XmlElement(required = false, name = "base64", type = Base64DataFormat.class),
084        @XmlElement(required = false, name = "beanio", type = BeanioDataFormat.class),
085        @XmlElement(required = false, name = "bindy", type = BindyDataFormat.class),
086        @XmlElement(required = false, name = "boon", type = BoonDataFormat.class),
087        @XmlElement(required = false, name = "castor", type = CastorDataFormat.class),
088        @XmlElement(required = false, name = "crypto", type = CryptoDataFormat.class),
089        @XmlElement(required = false, name = "csv", type = CsvDataFormat.class),
090        @XmlElement(required = false, name = "custom", type = CustomDataFormat.class),
091        @XmlElement(required = false, name = "flatpack", type = FlatpackDataFormat.class),
092        @XmlElement(required = false, name = "gzip", type = GzipDataFormat.class),
093        @XmlElement(required = false, name = "hl7", type = HL7DataFormat.class),
094        @XmlElement(required = false, name = "ical", type = IcalDataFormat.class),
095        @XmlElement(required = false, name = "jacksonxml", type = JacksonXMLDataFormat.class),
096        @XmlElement(required = false, name = "jaxb", type = JaxbDataFormat.class),
097        @XmlElement(required = false, name = "jibx", type = JibxDataFormat.class),
098        @XmlElement(required = false, name = "json", type = JsonDataFormat.class),
099        @XmlElement(required = false, name = "protobuf", type = ProtobufDataFormat.class),
100        @XmlElement(required = false, name = "rss", type = RssDataFormat.class),
101        @XmlElement(required = false, name = "secureXML", type = XMLSecurityDataFormat.class),
102        @XmlElement(required = false, name = "serialization", type = SerializationDataFormat.class),
103        @XmlElement(required = false, name = "soapjaxb", type = SoapJaxbDataFormat.class),
104        @XmlElement(required = false, name = "string", type = StringDataFormat.class),
105        @XmlElement(required = false, name = "syslog", type = SyslogDataFormat.class),
106        @XmlElement(required = false, name = "tarfile", type = TarFileDataFormat.class),
107        @XmlElement(required = false, name = "tidyMarkup", type = TidyMarkupDataFormat.class),
108        @XmlElement(required = false, name = "univocity-csv", type = UniVocityCsvDataFormat.class),
109        @XmlElement(required = false, name = "univocity-fixed", type = UniVocityFixedWidthDataFormat.class),
110        @XmlElement(required = false, name = "univocity-tsv", type = UniVocityTsvDataFormat.class),
111        @XmlElement(required = false, name = "xmlBeans", type = XMLBeansDataFormat.class),
112        @XmlElement(required = false, name = "xmljson", type = XmlJsonDataFormat.class),
113        @XmlElement(required = false, name = "xmlrpc", type = XmlRpcDataFormat.class),
114        @XmlElement(required = false, name = "xstream", type = XStreamDataFormat.class),
115        @XmlElement(required = false, name = "pgp", type = PGPDataFormat.class),
116        @XmlElement(required = false, name = "zip", type = ZipDataFormat.class),
117        @XmlElement(required = false, name = "zipFile", type = ZipFileDataFormat.class)}
118        )
119    private DataFormatDefinition dataFormatType;
120
121    @XmlAttribute
122    @Deprecated
123    private String ref;
124
125    public UnmarshalDefinition() {
126    }
127
128    public UnmarshalDefinition(DataFormatDefinition dataFormatType) {
129        this.dataFormatType = dataFormatType;
130    }
131
132    public UnmarshalDefinition(String ref) {
133        this.ref = ref;
134    }
135
136    @Override
137    public String toString() {
138        return "Unmarshal[" + description() + "]";
139    }
140
141    protected String description() {
142        if (dataFormatType != null) {
143            return dataFormatType.toString();
144        } else {
145            return "ref:" + ref;
146        }
147    }
148
149    @Override
150    public String getLabel() {
151        return "unmarshal[" + description() + "]";
152    }
153
154    public String getRef() {
155        return ref;
156    }
157
158    /**
159     * To refer to a custom data format to use as unmarshaller
160     *
161     * @deprecated use custom dataformat instead
162     */
163    @Deprecated
164    public void setRef(String ref) {
165        this.ref = ref;
166    }
167
168    public DataFormatDefinition getDataFormatType() {
169        return dataFormatType;
170    }
171
172    /**
173     * The data format to be used
174     */
175    public void setDataFormatType(DataFormatDefinition dataFormatType) {
176        this.dataFormatType = dataFormatType;
177    }
178
179    @Override
180    public Processor createProcessor(RouteContext routeContext) {
181        DataFormat dataFormat = DataFormatDefinition.getDataFormat(routeContext, getDataFormatType(), ref);
182        return new UnmarshalProcessor(dataFormat);
183    }
184}
