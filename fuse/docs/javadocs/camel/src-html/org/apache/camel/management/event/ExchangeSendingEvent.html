017package org.apache.camel.management.event;
018
019import org.apache.camel.Endpoint;
020import org.apache.camel.Exchange;
021
022/**
023 * Event for <b>before</b> sending an {@link Exchange} to an {@link Endpoint}.
024 * <p/>
025 * This event is emitted before attempting to send the {@link Exchange} to an {@link Endpoint}.
026 * There is still some internal processing occurring before the actual sending takes places, and
027 * therefore its not a 100% guarantee that the sending actually happens, as there may cause an
028 * internal error before.
029 * <p/>
030 * The {@link ExchangeSentEvent} is an event which is emitted <b>after</b> the sending is done.
031 *
032 * @see ExchangeSentEvent
033 * @version 
034 */
035public class ExchangeSendingEvent extends AbstractExchangeEvent {
036    private static final long serialVersionUID = -19248832613958122L;
037
038    private final Endpoint endpoint;
039
040    public ExchangeSendingEvent(Exchange source, Endpoint endpoint) {
041        super(source);
042        this.endpoint = endpoint;
043    }
044
045    public Endpoint getEndpoint() {
046        return endpoint;
047    }
048
049    @Override
050    public String toString() {
051        return getExchange().getExchangeId() + " exchange " + getExchange() + " sending to: " + endpoint;
052    }
053
054}
